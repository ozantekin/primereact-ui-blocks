Daha karmaşık JSX bileşen örneği:

```jsx
import React, { useState } from "react";

function Sayac() {
  const [sayi, setSayi] = useState(0);

  return (
    <div>
      <p>Sayac: {sayi}</p>
      <button onClick={() => setSayi(sayi + 1)}>Arttır</button>
      <button onClick={() => setSayi(sayi - 1)}>Azalt</button>
    </div>
  );
}

<Sayac />;
```
